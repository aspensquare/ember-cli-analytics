import Ember from 'ember'; export default Ember.Object.extend({

Method trackPage

Send the current page URL to the analytics engine.

trackPage: Ember.K,

Method trackEvent

Send an arbitrary event to the anlytics engine.

trackEvent: Ember.K,

Method trackConversion

Send a conversion completion event to the analytics engine.

trackConversion: Ember.K,

Method identity

Identify an anonymous user with a unique ID. This is useful when a user returns to the application an we wish to further track them.

This should not be called in conjunction with alias.

identify: Ember.K,

Method alias

For those platforms that support it, map an anonymous user id to a registered user. This is useful when you wish to associate events made before the user registerd with a newly created user account.

alias: Ember.K });
